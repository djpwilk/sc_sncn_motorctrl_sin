192 int update_statusword(int current_status, int state_reached, int ack, int q_active, int shutdown_ack) {
193  int status_word;
194 
195  switch (state_reached) {
196  case 1:
197  status_word = current_status & ~READY_TO_SWITCH_ON_STATE
198  & ~SWITCHED_ON_STATE & ~OPERATION_ENABLED_STATE
199  & ~VOLTAGE_ENABLED_STATE;
200  break;
201 
202  case 7:
203  status_word = (current_status & ~OPERATION_ENABLED_STATE
204  & ~SWITCHED_ON_STATE & ~VOLTAGE_ENABLED_STATE
205  & ~SWITCH_ON_DISABLED_STATE) | READY_TO_SWITCH_ON_STATE;
206  break;
207 
208  case 2:
209  status_word = (current_status & READY_TO_SWITCH_ON_STATE
210  & ~OPERATION_ENABLED_STATE & ~SWITCHED_ON_STATE
211  & ~VOLTAGE_ENABLED_STATE) | SWITCH_ON_DISABLED_STATE;
212  break;
213 
214  case 3:
215  status_word = (current_status & ~SWITCH_ON_DISABLED_STATE
216  & ~OPERATION_ENABLED_STATE) | SWITCHED_ON_STATE
217  | VOLTAGE_ENABLED_STATE;
218  break;
219 
220  case 4:
221  status_word = current_status | OPERATION_ENABLED_STATE;
222  break;
223 
224  case 5:
225  status_word = current_status | FAULT_STATE;
226  break;
227 
228  case 6:
229  status_word = current_status | QUICK_STOP_STATE;
230  break;
231 
232  }
233  if(q_active == 1)
234  return status_word & (~QUICK_STOP_STATE);
235  if(shutdown_ack == 1)
236  return status_word & (~VOLTAGE_ENABLED_STATE);
237  if(ack == 1)
