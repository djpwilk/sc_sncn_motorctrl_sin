85  while ( i < ADC_CALIB_POINTS)

87  /* get ADC reading */

89

92  c_adc :> a;
93  c_adc :> b;

95  if(a>0 && a<16384 && b>0 && b<16384)
96  {
97  I_calib.Ia_calib += a;
98  I_calib.Ib_calib += b;
99  i++;
100  if(i == ADC_CALIB_POINTS)
101  break;
102  }
103  }
104  I_calib.Ia_calib = (I_calib.Ia_calib >> Factor);
105  I_calib.Ib_calib = (I_calib.Ib_calib >> Factor);
