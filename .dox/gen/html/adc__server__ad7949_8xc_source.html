153 static inline unsigned convert(unsigned raw)
154 {
155  unsigned int data;
156 
157  /* raw == 0b xxxx aabb ccdd eeff ...
158  we read every data bit twice because of port clock setting */
159 
160  raw = bitrev(raw);
161  data = raw & 0x06000000;
162  data >>= 2;
163  data |= raw & 0x00600000;
164  data >>= 2;
165  data |= raw & 0x00060000;
166  data >>= 2;
167  data |= raw & 0x00006000;
168  data >>= 2;
169  data |= raw & 0x00000600;
170  data >>= 2;
171  data |= raw & 0x00000060;
172  data >>= 2;
173  data |= raw & 0x00000006;
174  data >>= 1;
175  return data;
176 }

178 /*
179  * ADC CONFIGURAION HOWTO
180  *
181  * Serialization: LSB first
182  *
183  * adc_config: !! reverse bit-order compared to AD7949 datasheet !!
184  * Bit Name Description
185  * 32:14 0 unused
186  * 13 RB Read back CFG reg ( 1 => no)
187  * 12:11 SEQ Sequencer ( 00 => disable)
188  * 10:8 REF Reference selection (100 => internal)
189  * 7 BW LPF bandwidth ( 1 => full)
190  * 6:4 INx Input channel selection (000 => CH0)
191  * 3:1 INCC Input channel config (111 => unipolar, referenced to GND)
192  * 0 CFG Config update ( 1 => overwrite cfg reg)
193  * -1 --PADDING-- (delay 1 clk cycle)
194  * => adc_config = 0b100100100011110
195 */

198 static void adc_ad7949_singleshot(buffered out port:32 p_sclk_conv_mosib_mosia,
199  in buffered port:32 p_data_a,
200  in buffered port:32 p_data_b,
201  clock clk,
202  const unsigned int adc_config_mot,
203  const unsigned int adc_config_other[],
204  const unsigned int delay,
205  timer t)
206 {
207 
208  unsigned int ts;
209  unsigned int data_raw_a;
210  unsigned int data_raw_b;
211 
212  /* Reading/Writing after conversion (RAC)
213  Read previous conversion result
214  Write CFG for next conversion */
215  #define SPI_IDLE configure_out_port(p_sclk_conv_mosib_mosia, clk, 0b0100)
216 
217  // CONGIG__other_n1 CFG_Imotx_x1 |CONGIG__other_n2 CFG_Imotx_x2 |CONGIG__other_n3 CFG_Imotx_x3 |
218  // CONVERT_null CONVERT_other_n1 |CONVERT_Imot_x1 CONVERT_other_n2 |CONVERT_Imot_x2 CONVERT_other_n3 |
219  // READOUT_null READOUT_other_null |READOUT_other_n1 READOUT_Imot_x1 |READOUT_other_n2 READOUT_Imot_x2 |
220  // -----------
221  // iIndexADC 0 1 2 3
222  // readout extern temperature current-voltage extern
223 
224 
225  SPI_IDLE;
226 // p_ifm_ext_d3 <: 1;
227  output_adc_config_data(clk, p_data_a, p_data_b, p_sclk_conv_mosib_mosia, adc_config_other[adc_index]);
228  SPI_IDLE;
229 
230  t :> ts;
231  p_data_a :> data_raw_a;
232  adc_data_a[adc_index] = convert(data_raw_a);
233  p_data_b :> data_raw_b;
234  adc_data_b[adc_index] = convert(data_raw_b);
235  adc_index++;
236  adc_index &= 0x3;
237  t when timerafter(ts + delay) :> ts;
238 
239  output_adc_config_data(clk, p_data_a, p_data_b, p_sclk_conv_mosib_mosia, adc_config_mot);
240  SPI_IDLE;
241 // p_ifm_ext_d3 <: 0;
242 
243  p_data_a :> data_raw_a;
244  adc_data_a[4] = convert(data_raw_a);
245  p_data_b :> data_raw_b;
246  adc_data_b[4] = convert(data_raw_b);
247 
248  SPI_IDLE;
249  t :> ts;
250 }
