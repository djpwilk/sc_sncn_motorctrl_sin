49 {

57 
58  if(AbsX <0) AbsX = -AbsX;
59  if(AbsY <0) AbsY = -AbsY;
60 
61  #define defFactor 65536/64
62 
63  if(AbsY < AbsX)
64  {
65  cFlagOver45 = 0;
66  AbsY *= defFactor;
67  if(AbsX==0) Mag = defFactor;
68  else
69  {
70  if (AbsY==0) Mag = 0;
71  else Mag = AbsY / AbsX;
72  }
73  }
74  else
75  {
76  cFlagOver45 = 1;
77  AbsX *= defFactor;
78  if(AbsY==0) Mag = defFactor;

80  {
81  if (AbsX==0) Mag = 0;
82  else Mag = AbsX / AbsY;
83  }
84  }
85

87  iAngleResult = arctg_table[Mag];
88

92  { // 1. or 4. quadrant

95  }
96  else

98  if(Imag < 0) iAngleResult += 2048; // 3. Q
99  else iAngleResult = 2048 - iAngleResult; // 2. Quadrant
