212 void commutation_init_ecat(chanend c_signal, hall_par &hall_params, qei_par &qei_params, commutation_par &commutation_params)
213 {
214  int command;
215  int nominal_speed;
216  int flag = 0;
217  int init_state = INIT_BUSY;
218  while(1)
219  {
220  #pragma ordered
221  select
222  {
223  case c_signal :> command:
224 
225  if(command == CHECK_BUSY)
226  {
227  c_signal <: init_state;
228  }
229  else if(command == SET_COMM_PARAM_ECAT)
230  {
231  c_signal :> hall_params.gear_ratio;
232  c_signal :> hall_params.pole_pairs;
233  c_signal :> qei_params.gear_ratio;
234  c_signal :> qei_params.index;
235  c_signal :> qei_params.max_count;
236  c_signal :> qei_params.real_counts;
237  c_signal :> nominal_speed;
238  flag = 1;
239 
240  commutation_params.angle_variance = 1024/(hall_params.pole_pairs * 3);
241  if(hall_params.pole_pairs < 4)
242  {
243  commutation_params.max_speed_reached = nominal_speed*4;
244  commutation_params.flag = 1;
245  }
246  else if(hall_params.pole_pairs >=4)
247  {
248  commutation_params.max_speed_reached = nominal_speed;
249  commutation_params.flag = 0;
250  }
251  commutation_params.qei_forward_offset = 0;
252  commutation_params.qei_backward_offset = 0;
253 
254 
255 
256  }
257  break;
258 
259  default:
260  break;
261 
262  }
263  if(flag == 1)
264  break;
265  }
